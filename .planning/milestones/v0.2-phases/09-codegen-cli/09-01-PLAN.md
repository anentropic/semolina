## Overview

Set up the Typer CLI foundation: entry point, command structure, argument/option
parsing, backend enum, help text, and version display. This is the shell of the
`cubano codegen` command — it parses and validates all user input but delegates
actual SQL generation to stubs that Wave 2 plans will implement.

## Goal

Running `cubano codegen <input> --backend snowflake` exits 0 when the input path
exists, routes all diagnostics to stderr and any SQL to stdout, and shows
meaningful help text via `--help`. The `--version` flag shows the Cubano version.
All argument validation (required flags, enum values, path existence) works
correctly and returns exit code 1 with a clear error message on failure.

### Task 3: Create `src/cubano/cli/codegen.py` — codegen subcommand

This is the primary CLI surface. The command must:

- Accept a required positional argument `input` (file, directory, or glob pattern)
- Accept a required `--backend`/`-b` option (enum: `snowflake` or `databricks`)
- Accept an optional `--output`/`-o` option (path; default None = stdout)
- Accept `--verbose`/`-v` flag (bool; default False)
- Accept `--no-comments` flag (bool; default False — include docstring comments by default)
- Route all progress/status messages to stderr via Rich Console
- Route generated SQL to stdout (or the `--output` file)
- Fail immediately (exit 1) if input path does not exist or is not readable
- Process directories by scanning for `*.py` files recursively
- Expand glob patterns using `pathlib.Path.glob()`
- Silently skip files that contain no SemanticView subclasses

```python
"""
Codegen subcommand: generate warehouse-native SQL from Cubano models.
"""

from __future__ import annotations

import sys
from enum import Enum
from pathlib import Path
from typing import Annotated

import typer
from rich.console import Console

codegen_app = typer.Typer(
    name="codegen",
    help="Generate warehouse-native CREATE SEMANTIC VIEW SQL from Cubano model files.",
    no_args_is_help=True,
)

# Separate consoles: SQL to stdout, diagnostics to stderr
_stdout = Console(file=sys.stdout, stderr=False, highlight=False)
_stderr = Console(file=sys.stderr, stderr=True)


class Backend(str, Enum):
    """Target warehouse backend for SQL generation."""
    snowflake = "snowflake"
    databricks = "databricks"


@codegen_app.command()
def codegen(
    input: Annotated[
        str,
        typer.Argument(help="Model file, directory, or glob pattern (e.g. src/models.py, src/, 'src/**/*.py')"),
    ],
    backend: Annotated[
        Backend,
        typer.Option("--backend", "-b", help="Target warehouse backend."),
    ],
    output: Annotated[
        Path | None,
        typer.Option("--output", "-o", help="Output file path. Defaults to stdout."),
    ] = None,
    verbose: Annotated[
        bool,
        typer.Option("--verbose", "-v", help="Enable verbose diagnostic output."),
    ] = False,
    no_comments: Annotated[
        bool,
        typer.Option("--no-comments", help="Strip docstring comments from generated SQL."),
    ] = False,
) -> None:
    """Generate warehouse-native SQL from Cubano model definitions."""
    from cubano.cli.utils import make_stderr_console, resolve_input_paths
    from cubano.codegen.generator import generate_sql_for_files

    stderr = make_stderr_console(verbose=verbose)

    # Resolve input to a list of .py file paths
    try:
        file_paths = resolve_input_paths(input, stderr=stderr)
    except SystemExit:
        raise
    except Exception as e:
        _stderr.print(f"[bold red]Error:[/bold red] {e}")
        raise typer.Exit(code=1)

    if not file_paths:
        _stderr.print(
            f"[yellow]Warning:[/yellow] No Python files found matching [cyan]{input!r}[/cyan]."
        )
        raise typer.Exit(code=0)

    if verbose:
        stderr.print(
            f"[dim]Found {len(file_paths)} file(s) to process.[/dim]"
        )

    # Generate SQL (stub: implemented in 09-03)
    sql_blocks = generate_sql_for_files(
        file_paths=file_paths,
        backend=backend.value,
        include_comments=not no_comments,
        stderr=stderr,
        verbose=verbose,
    )

    # Write output
    combined_sql = "\n\n".join(sql_blocks)

    if output is not None:
        output.write_text(combined_sql, encoding="utf-8")
        _stderr.print(
            f"[green]✓[/green] Generated SQL written to [cyan]{output}[/cyan]"
        )
    else:
        _stdout.print(combined_sql)
```

### Task 4: Create `src/cubano/cli/utils.py` — shared CLI helpers

```python
"""
CLI utility functions for Cubano codegen commands.

Provides path resolution, stderr console creation, and progress helpers.
"""

from __future__ import annotations

import sys
from pathlib import Path

import typer
from rich.console import Console


def make_stderr_console(verbose: bool = False) -> Console:
    """
    Create a Rich Console writing to stderr for diagnostic output.

    Args:
        verbose: If True, suppress no output (show everything). If False, only
            show warnings and errors.

    Returns:
        Configured Rich Console for stderr.
    """
    return Console(file=sys.stderr, stderr=True)


def resolve_input_paths(input_spec: str, stderr: Console) -> list[Path]:
    """
    Resolve an input specification to a list of Python file paths.

    Handles three forms:
    - Explicit file path: must exist, returned as single-element list
    - Directory path: scanned recursively for *.py files
    - Glob pattern: expanded via pathlib from cwd

    Fails immediately (raises SystemExit via typer.Exit) if an explicit
    file path doesn't exist or can't be read.

    Args:
        input_spec: File path, directory path, or glob pattern string.
        stderr: Rich Console for diagnostic messages.

    Returns:
        Sorted list of resolved .py file paths.

    Raises:
        typer.Exit: If an explicit path doesn't exist or is unreadable.
    """
    path = Path(input_spec)

    # Explicit file: must exist
    if path.suffix == ".py":
        if not path.exists():
            stderr.print(
                f"[bold red]Error:[/bold red] File not found: [cyan]{path}[/cyan]\n"
                f"  Check the path and try again."
            )
            raise typer.Exit(code=1)
        if not path.is_file():
            stderr.print(
                f"[bold red]Error:[/bold red] Not a file: [cyan]{path}[/cyan]"
            )
            raise typer.Exit(code=1)
        return [path.resolve()]

    # Directory: scan recursively for *.py files
    if path.is_dir():
        files = sorted(path.rglob("*.py"))
        # Exclude __pycache__ and hidden directories
        files = [
            f for f in files
            if "__pycache__" not in f.parts and not any(p.startswith(".") for p in f.parts)
        ]
        return files

    # Explicit directory path that doesn't exist: fail immediately
    if not path.suffix and not path.exists():
        stderr.print(
            f"[bold red]Error:[/bold red] Directory not found: [cyan]{path}[/cyan]\n"
            f"  Check the path and try again."
        )
        raise typer.Exit(code=1)

    # Glob pattern: expand from current working directory
    from pathlib import Path as P
    cwd = P.cwd()
    matched = sorted(cwd.glob(input_spec))
    py_files = [f for f in matched if f.suffix == ".py" and f.is_file()]
    return py_files
```
