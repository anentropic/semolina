## Custom env_file Parameter Usage

```bash
# Test 1: Default behavior (backward compatible)
export SNOWFLAKE_ACCOUNT=test_account
export SNOWFLAKE_USER=test_user
export SNOWFLAKE_PASSWORD=test_password
export SNOWFLAKE_WAREHOUSE=test_warehouse
export SNOWFLAKE_DATABASE=test_db
unset CUBANO_ENV_FILE
uv run python -c "from cubano.testing.credentials import SnowflakeCredentials; c = SnowflakeCredentials.load(); print('Default path works')"
# Should print: Default path works (uses .env by default)

# Test 2: Custom env_file parameter
echo "SNOWFLAKE_ACCOUNT=custom_account" > /tmp/custom.env
echo "SNOWFLAKE_USER=custom_user" >> /tmp/custom.env
echo "SNOWFLAKE_PASSWORD=custom_pass" >> /tmp/custom.env
echo "SNOWFLAKE_WAREHOUSE=custom_wh" >> /tmp/custom.env
echo "SNOWFLAKE_DATABASE=custom_db" >> /tmp/custom.env
uv run python -c "from cubano.testing.credentials import SnowflakeCredentials; c = SnowflakeCredentials.load(env_file='/tmp/custom.env'); print(f'Loaded from custom path: {c.account}')"
# Should print: Loaded from custom path: custom_account

# Test 3: CUBANO_ENV_FILE environment variable (highest priority)
export CUBANO_ENV_FILE=/tmp/custom.env
echo "SNOWFLAKE_ACCOUNT=env_var_account" > /tmp/env_override.env
export SNOWFLAKE_PASSWORD=env_var_pass
export SNOWFLAKE_USER=env_var_user
export SNOWFLAKE_WAREHOUSE=env_var_wh
export SNOWFLAKE_DATABASE=env_var_db
uv run python -c "from cubano.testing.credentials import SnowflakeCredentials; c = SnowflakeCredentials.load(env_file='/tmp/other.env'); print(f'Loaded from CUBANO_ENV_FILE: {c.account}')"
# Should print: Loaded from CUBANO_ENV_FILE: custom_account (because /tmp/custom.env is used, not /tmp/other.env)
```
