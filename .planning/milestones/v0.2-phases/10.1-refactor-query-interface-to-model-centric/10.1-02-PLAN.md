   ```python
   def __eq__(self, value: Any) -> Q:
       """
       Create equality filter: field == value

       Returns a Q object with this field set to the given value.

       Example:
           >>> Users.country == 'US'
           <Q object>
       """
       if self.name is None:
           raise RuntimeError("Field operator used before __set_name__ called")
       return Q(**{self.name: value})

   def __ne__(self, value: Any) -> Q:
       """
       Create inequality filter: field != value

       Returns negated Q object.
       """
       if self.name is None:
           raise RuntimeError("Field operator used before __set_name__ called")
       return ~Q(**{self.name: value})

   def __lt__(self, value: Any) -> Q:
       """
       Create less-than filter: field < value

       Uses __lt suffix convention for warehouse compatibility.
       """
       if self.name is None:
           raise RuntimeError("Field operator used before __set_name__ called")
       return Q(**{f"{self.name}__lt": value})

   def __le__(self, value: Any) -> Q:
       """
       Create less-than-or-equal filter: field <= value
       """
       if self.name is None:
           raise RuntimeError("Field operator used before __set_name__ called")
       return Q(**{f"{self.name}__le": value})

   def __gt__(self, value: Any) -> Q:
       """
       Create greater-than filter: field > value
       """
       if self.name is None:
           raise RuntimeError("Field operator used before __set_name__ called")
       return Q(**{f"{self.name}__gt": value})

   def __ge__(self, value: Any) -> Q:
       """
       Create greater-than-or-equal filter: field >= value
       """
       if self.name is None:
           raise RuntimeError("Field operator used before __set_name__ called")
       return Q(**{f"{self.name}__ge": value})

   # Preserve identity-based hashing even with __eq__ defined
   # (Python sets __hash__ = None when __eq__ is defined without explicit __hash__)
   __hash__ = object.__hash__
   ```

   ```python
   def where(self, condition: Q | None) -> Query:
       """
       Add WHERE filter condition to query.

       Filters rows based on boolean Q object conditions. Multiple calls
       are ANDed together. Use Q composition (|, &, ~) for complex logic.

       Args:
           condition: Q object or None (None is no-op)

       Returns:
           New Query with condition added

       Raises:
           TypeError: If condition is not a Q object or None

       Example:
           >>> query = Query()
           >>> q1 = query.where(Users.country == 'US')
           >>> q2 = q1.where(Users.revenue > 1000)
       """
       if condition is None:
           return self

       if not isinstance(condition, Q):
           raise TypeError(
               f"where() requires a Q object, got {type(condition).__name__}. "
               f"Use: .where(Field == value) or .where(Q(...) & Q(...))"
           )

       # Combine with existing filters using AND
       new_filter = condition if self._filters is None else self._filters & condition
       return replace(self, _filters=new_filter)
   ```
