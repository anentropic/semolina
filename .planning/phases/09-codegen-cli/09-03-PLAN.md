## Overview

Implement the module introspection layer: dynamic loading of user Python files
via `importlib.util`, extraction of `SemanticView` subclasses using `inspect`,
Python syntax validation via `ast.parse`, and error collection using
`ExceptionGroup`. This plan replaces the stub `generator.py` from 09-01 with
a real implementation that wires together loading, introspection, validation,
and rendering.

## Goal

`generate_sql_for_files(file_paths, backend, include_comments, stderr, verbose)`
returns a list of rendered SQL/YAML strings. Files without SemanticView models
are silently skipped. Python syntax errors and import failures are collected and
reported as a group (not fail-fast). The function emits Rich-formatted progress
feedback to stderr, including a progress bar when processing more than 5 files.

### Task 1: Create `src/cubano/codegen/loader.py` — dynamic module loading and introspection

```python
"""
Dynamic module loading and SemanticView introspection for Cubano codegen.

Provides functions to load Python modules from arbitrary file paths
and extract SemanticView subclass metadata.
"""

from __future__ import annotations

import ast
import importlib.util
import inspect
import sys
from pathlib import Path

from cubano.codegen.models import FieldData, ModelData
from cubano.fields import Dimension, Fact, Metric
from cubano.models import SemanticView


def validate_python_syntax(file_path: Path) -> None:
    """
    Validate Python file syntax using ast.parse before dynamic import.

    This catches syntax errors early with precise line numbers, before
    importlib attempts to execute the file.

    Args:
        file_path: Path to the Python source file to validate.

    Raises:
        SyntaxError: If the file contains invalid Python syntax. The error
            includes filename, line number, and offset for precise reporting.
    """
    source = file_path.read_text(encoding="utf-8")
    # ast.parse raises SyntaxError with filename/lineno set if invalid
    ast.parse(source, filename=str(file_path))


def load_module_from_path(file_path: Path) -> object:
    """
    Dynamically load a Python module from an arbitrary file path.

    Uses importlib.util to load the module without requiring it to be on
    sys.path. The module is cached in sys.modules using a unique key derived
    from the file path to prevent duplicate imports.

    Args:
        file_path: Absolute path to the Python source file.

    Returns:
        Loaded module object.

    Raises:
        ImportError: If the spec cannot be created or the loader is missing.
        Exception: Any exception raised during module execution propagates.
    """
    # Use an absolute, normalised path as the module name to ensure uniqueness
    module_name = f"_cubano_codegen_{file_path.resolve().as_posix().replace('/', '_').lstrip('_')}"

    spec = importlib.util.spec_from_file_location(module_name, file_path)
    if spec is None or spec.loader is None:
        raise ImportError(
            f"Cannot create module spec for {file_path}. "
            f"Ensure the file is a valid Python source file."
        )

    module = importlib.util.module_from_spec(spec)
    # Cache before exec so circular imports within the module work
    sys.modules[module_name] = module
    try:
        spec.loader.exec_module(module)  # type: ignore[union-attr]
    except Exception:
        # Remove from cache on failure to allow retries
        sys.modules.pop(module_name, None)
        raise

    return module


def extract_models_from_module(module: object, source_file: Path) -> list[ModelData]:
    """
    Extract all SemanticView subclasses from a loaded module.

    Uses inspect.getmembers() to find all classes in the module that are
    subclasses of SemanticView (but not SemanticView itself). Only classes
    defined in the module being processed are included (not re-exported ones).

    Args:
        module: Loaded Python module object.
        source_file: Path to the source file (for ModelData.source_file).

    Returns:
        List of ModelData objects, one per SemanticView subclass found.
        Empty list if no SemanticView subclasses are in the module.
    """
    models: list[ModelData] = []

    for _name, obj in inspect.getmembers(module, inspect.isclass):
        # Must be a SemanticView subclass, not SemanticView itself
        if not (isinstance(obj, type) and issubclass(obj, SemanticView) and obj is not SemanticView):
            continue

        # Only process classes defined in this module (not imported ones)
        if getattr(obj, "__module__", None) != getattr(module, "__name__", None):
            continue

        model = _class_to_model_data(obj, source_file)
        models.append(model)

    return models


def _class_to_model_data(cls: type, source_file: Path) -> ModelData:
    """
    Convert a SemanticView subclass to a ModelData dataclass.

    Extracts the view name, class docstring, and all field names/types/docstrings
    from the class's _fields MappingProxy.

    Args:
        cls: SemanticView subclass.
        source_file: Path to the source file containing this class.

    Returns:
        ModelData instance with all extracted metadata.
    """
    metrics: list[FieldData] = []
    dimensions: list[FieldData] = []
    facts: list[FieldData] = []

    # cls._fields is a MappingProxyType({name: Field}) in definition order
    for field_name, field_obj in cls._fields.items():  # type: ignore[attr-defined]
        docstring = inspect.getdoc(field_obj) or ""
        if isinstance(field_obj, Metric):
            metrics.append(FieldData(name=field_name, field_type="metric", docstring=docstring))
        elif isinstance(field_obj, Dimension):
            dimensions.append(FieldData(name=field_name, field_type="dimension", docstring=docstring))
        elif isinstance(field_obj, Fact):
            facts.append(FieldData(name=field_name, field_type="fact", docstring=docstring))

    return ModelData(
        class_name=cls.__name__,
        view_name=cls._view_name,  # type: ignore[attr-defined]
        docstring=inspect.getdoc(cls) or "",
        metrics=metrics,
        dimensions=dimensions,
        facts=facts,
        source_file=str(source_file.resolve()),
    )
```

### Task 2: Create `src/cubano/codegen/validator.py` — input validation and error collection

This module handles per-file validation and formats errors for human display.

```python
"""
Input validation and error collection for Cubano codegen.

Provides functions to validate Python model files and collect errors
for comprehensive multi-error reporting.
"""

from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path


@dataclass
class CodegenError:
    """
    A single codegen error with context for human-readable reporting.

    Attributes:
        file_path: Path to the file where the error occurred.
        message: Human-readable description of the problem.
        hint: Optional suggestion for how to fix the error.
        line: Optional line number where the error occurred.
    """

    file_path: Path
    message: str
    hint: str = ""
    line: int | None = None

    def format(self) -> str:
        """
        Format the error as a single-line diagnostic message.

        Returns:
            Formatted string like: "models.py:12: SyntaxError: invalid syntax"
        """
        loc = str(self.file_path)
        if self.line is not None:
            loc = f"{loc}:{self.line}"
        parts = [f"  {loc}: {self.message}"]
        if self.hint:
            parts.append(f"    Hint: {self.hint}")
        return "\n".join(parts)


def format_error_report(errors: list[CodegenError]) -> str:
    """
    Format a list of CodegenErrors into a human-readable report.

    Args:
        errors: List of errors collected during processing.

    Returns:
        Multi-line string with all errors and hints.
    """
    lines = [f"Codegen failed with {len(errors)} error(s):"]
    for err in errors:
        lines.append(err.format())
    return "\n".join(lines)
```

### Task 3: Replace the stub `src/cubano/codegen/generator.py` with the real implementation

The real `generator.py` wires together: path resolution → syntax validation →
module loading → model extraction → template rendering → output assembly.

```python
"""
SQL generation orchestrator for Cubano codegen.

Wires together module loading, introspection, and template rendering.
Collects errors from all files and reports them together.
"""

from __future__ import annotations

from pathlib import Path

import typer
from rich.console import Console
from rich.progress import BarColumn, Progress, SpinnerColumn, TextColumn, TimeElapsedColumn

from cubano.codegen.loader import (
    extract_models_from_module,
    load_module_from_path,
    validate_python_syntax,
)
from cubano.codegen.renderer import render_file_header, render_view
from cubano.codegen.validator import CodegenError, format_error_report

# Show progress bar when processing more than this many files
_PROGRESS_THRESHOLD = 5


def generate_sql_for_files(
    file_paths: list[Path],
    backend: str,
    include_comments: bool,
    stderr: Console,
    verbose: bool,
) -> list[str]:
    """
    Generate SQL for all SemanticView models in the given files.

    Processing pipeline per file:
    1. Validate Python syntax (ast.parse)
    2. Load module (importlib.util)
    3. Extract SemanticView subclasses (inspect)
    4. Render each model (Jinja2 template)
    5. Collect errors; report at end

    Args:
        file_paths: List of Python source files to process.
        backend: Target backend name ('snowflake' or 'databricks').
        include_comments: Whether to include docstring comments in output.
        stderr: Rich Console for diagnostic output (progress, errors).
        verbose: Whether to emit per-file verbose diagnostics.

    Returns:
        List of rendered SQL/YAML strings, one per input file that
        contained at least one SemanticView model.

    Raises:
        typer.Exit: With code 1 if any files produced errors.
    """
    errors: list[CodegenError] = []
    results: list[str] = []

    use_progress = len(file_paths) > _PROGRESS_THRESHOLD

    if use_progress:
        progress_ctx = Progress(
            SpinnerColumn(),
            TextColumn("[progress.description]{task.description}"),
            BarColumn(),
            TextColumn("{task.completed}/{task.total}"),
            TimeElapsedColumn(),
            console=stderr,
        )
    else:
        # Null context manager — no progress bar for small file sets
        from contextlib import nullcontext
        progress_ctx = nullcontext()  # type: ignore[assignment]

    with progress_ctx as progress:
        task_id = None
        if use_progress and progress is not None:
            task_id = progress.add_task(
                f"[cyan]Generating {backend} SQL[/cyan]",
                total=len(file_paths),
            )

        for file_path in file_paths:
            if verbose:
                stderr.print(f"[dim]  Processing {file_path}[/dim]")

            try:
                # Step 1: Python syntax validation
                validate_python_syntax(file_path)

                # Step 2: Dynamic module loading
                module = load_module_from_path(file_path)

                # Step 3: Extract SemanticView subclasses
                models = extract_models_from_module(module, file_path)

                # Step 4: Silently skip files with no models
                if not models:
                    if verbose:
                        stderr.print(
                            f"[dim]    Skipped {file_path} — no SemanticView models found.[/dim]"
                        )
                    continue

                # Step 5: Render each model
                rendered_parts: list[str] = [render_file_header(str(file_path), backend)]
                for model in models:
                    rendered_parts.append(render_view(model, backend, include_comments))

                results.append("\n".join(rendered_parts))

            except SyntaxError as e:
                errors.append(
                    CodegenError(
                        file_path=file_path,
                        message=f"SyntaxError: {e.msg}",
                        hint="Fix the Python syntax error before running codegen.",
                        line=e.lineno,
                    )
                )
            except ImportError as e:
                errors.append(
                    CodegenError(
                        file_path=file_path,
                        message=f"ImportError: {e}",
                        hint=(
                            "Ensure all imports in this file are available in the "
                            "current Python environment."
                        ),
                    )
                )
            except Exception as e:
                errors.append(
                    CodegenError(
                        file_path=file_path,
                        message=f"{type(e).__name__}: {e}",
                        hint="Check the file for runtime errors at module level.",
                    )
                )
            finally:
                if use_progress and progress is not None and task_id is not None:
                    progress.advance(task_id)

    if errors:
        stderr.print(f"\n[bold red]{format_error_report(errors)}[/bold red]")
        raise typer.Exit(code=1)

    return results
```

Pay attention to:
- `spec.loader.exec_module(module)` — basedpyright may flag `union-attr` if the
  loader type is not narrowed. Use `# type: ignore[union-attr]` only after
  the `if spec.loader is None` guard makes a runtime check redundant.
- `cls._fields` access — since `SemanticView` declares `_fields: ClassVar[MappingProxyType]`,
  basedpyright should handle this. If it complains about `attr-defined`, add
  `# type: ignore[attr-defined]` with a comment explaining it's a ClassVar.
- `progress_ctx` assignment to `nullcontext()` — strict typing may require
  a union type annotation or Protocol. Use `# type: ignore[assignment]` with a comment.

## Verification Criteria

1. `validate_python_syntax(bad_file)` raises `SyntaxError` with line number
2. `load_module_from_path(model_file)` returns a module with the expected classes

   does not write any output for the good file (all-or-nothing error reporting)
6. `generate_sql_for_files([f], "snowflake", ...)` output contains `CREATE OR REPLACE SEMANTIC VIEW`
7. `generate_sql_for_files([f], "databricks", ...)` output contains YAML metric view structure
