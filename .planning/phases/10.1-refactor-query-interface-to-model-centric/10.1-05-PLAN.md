   ```python
   def test_model_centric_workflow_complete(self):
       """
       Demonstrate complete Phase 10.1 workflow:
       - Model definition with introspection
       - Query via model.query()
       - Field operators for filtering
       - Eager execution with Result
       """
       # 1. Define model
       class Sales(SemanticView, view='sales'):
           revenue = Metric()
           users_count = Metric()
           region = Dimension()
           country = Dimension()

       # 2. Introspect fields
       metrics = Sales.metrics()
       self.assertEqual(len(metrics), 2)
       self.assertEqual([m.name for m in metrics], ['revenue', 'users_count'])

       dims = Sales.dimensions()
       self.assertEqual(len(dims), 2)

       # 3. Build query with field operators
       result = (Sales.query()
           .metrics(Sales.revenue, Sales.users_count)
           .dimensions(Sales.region)
           .where((Sales.country == 'US') | (Sales.country == 'CA'))
           .where(Sales.revenue > 1000)
           .order_by(Sales.revenue.desc())
           .limit(10)
           .execute())

       # 4. Verify Result object
       self.assertIsInstance(result, Result)
       self.assertGreater(len(result), 0)

       # 5. Access rows
       for row in result:
           self.assertIn('revenue', row._data)
           self.assertIn('region', row._data)
           # Verify row access patterns
           _ = row.revenue
           _ = row['region']
   ```
